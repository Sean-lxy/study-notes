#### `package.json`

```json

{
  // 软件包的名称
  “name”: "sean",
  // 软件包的作者
  "author": "lxy",
  // 除作者外的贡献者
  "contributors": [],
  “bugs”: "",
  "homepage": "",
  "version": "",
  "license": "",
  "keywords": "",
  "description": "",
  "repository": "",
  "main": "",
  // 如果设置为 true，则可以防止应用程序/软件包被意外地发布到 npm
  "private": true,
  // 定义了一组可以运行的 node 脚本。
  "scripts": {},
	// 指定软件能运行的node和pnpm版本
	"engines": {
  	"node": ">=10",
    "pnpm": ">=3"
  },
  // 用于告知要支持哪些浏览器（及其版本）。
  "browserslist": ["> 1%", "last 2 versions", "not ie <= 8"]
}
```
